Poison Icons — Showcase

A neumorphic Material-Icons button builder — interactive preview and usage examples

Install
npm install poison-icons
or with yarn
yarn add poison-icons
Quick usage
import { IconButton, IconGallery } from "poison-icons";

// <IconButton icon="favorite" color="#f1b7a3" size={80} />

How to use (React + TypeScript)

// App.tsx
import React, { useState } from "react";
import { IconButton, IconGallery, ICONS } from "poison-icons";

export default function App() {
  const [pressedIcons, setPressedIcons] = useState(Array(ICONS.length).fill(false));

  const handleIconClick = (_icon: string, idx: number) => {
    setPressedIcons(prev => prev.map((pressed, i) => (i === idx ? !pressed : pressed)));
  };

  return (
    <div style={{ minHeight: "100vh", background: "#ECEEF1", padding: 32, boxSizing: "border-box" }}>
      <h1>Poison Icons Demo</h1>
      <IconButton icon="favorite" color="#f1b7a3" size={80} />

      <IconGallery
        color="#ECEEF1"
        size={64}
        onIconClick={(icon, idx) => handleIconClick(icon, idx)}
        pressedIcons={pressedIcons}
        showNames={true}
      />
    </div>
  );
}
